-- Kill Aura Toggle
HuntersTab:CreateToggle({
    Name = "Kill Aura",
    CurrentValue = false,
    Flag = "KillAuraToggle",
    Callback = function(Value)
        local range = 85
        local killInterval = 0.1
        local character = player.Character or player.CharacterAdded:Wait()
        local hrp = character:WaitForChild("HumanoidRootPart")

        local function killMob(mob)
            local humanoid = mob:FindFirstChild("Humanoid")
            if humanoid and humanoid.Health > 0 then
                humanoid.Health = 0
            end
        end

        task.spawn(function()
            while Value do
                wait(killInterval)
                local mobsFolder = workspace:FindFirstChild("Mobs")
                if mobsFolder then
                    for _, mob in pairs(mobsFolder:GetChildren()) do
                        if mob:FindFirstChild("HumanoidRootPart") and mob ~= character then
                            local distance = (hrp.Position - mob.HumanoidRootPart.Position).Magnitude
                            if distance <= range then
                                killMob(mob)
                            end
                        end
                    end
                end
            end
        end)
    end
})

-- Fire ProximityPrompt function
local function fireProximityPrompt(prompt)
    if prompt and prompt:IsA("ProximityPrompt") and prompt.Enabled then
        prompt:InputHoldBegin()
        task.wait(prompt.HoldDuration or 1)
        prompt:InputHoldEnd()
    end
end

local function fireAllProximityPrompts(plantsFolder)
    for _, plant in pairs(plantsFolder:GetChildren()) do
        local fruitsFolder = plant:FindFirstChild("Fruits")
        if fruitsFolder then
            for _, item in pairs(fruitsFolder:GetDescendants()) do
                if item:IsA("ProximityPrompt") and item.Enabled then
                    fireProximityPrompt(item)
                end
            end
        end
    end
end

local holdingE = false
local inputBeganConn, inputEndedConn, mouseClickConn

SettingsTab:CreateToggle({
    Name = "Teleport (Hold E + Click)",
    CurrentValue = false,
    Flag = "TeleportToggle",
    Callback = function(Value)
        if Value then
            inputBeganConn = UserInputService.InputBegan:Connect(function(input, processed)
                if not processed and input.KeyCode == Enum.KeyCode.E then
                    holdingE = true
                end
            end)
            inputEndedConn = UserInputService.InputEnded:Connect(function(input)
                if input.KeyCode == Enum.KeyCode.E then
                    holdingE = false
                end
            end)
            mouseClickConn = mouse.Button1Down:Connect(function()
                if not holdingE then return end
                local target = mouse.Hit and mouse.Hit.p
                local root = player.Character and player.Character:FindFirstChild("HumanoidRootPart")
                if target and root then
                    root.CFrame = CFrame.new(target + Vector3.new(0,3,0))
                end
            end)
        else
            if inputBeganConn then inputBeganConn:Disconnect() inputBeganConn = nil end
            if inputEndedConn then inputEndedConn:Disconnect() inputEndedConn = nil end
            if mouseClickConn then mouseClickConn:Disconnect() mouseClickConn = nil end
            holdingE = false
        end
    end
})

local isAutoFavoring = false
local favoredItems = {} -- track items already favored

-- Helper function to check if color matches
local function isTargetColor(color)
    -- Yellow: [253, 253, 31]
    if math.abs(color.R*255 - 253) <= 1 and
       math.abs(color.G*255 - 253) <= 1 and
       math.abs(color.B*255 - 31) <= 1 then
        return true
    end
    -- Red: [253, 31, 31]
    if math.abs(color.R*255 - 253) <= 1 and
       math.abs(color.G*255 - 31) <= 1 and
       math.abs(color.B*255 - 31) <= 1 then
        return true
    end
    return false
end

local function favorItem(itemSlot)
    local itemID = itemSlot.Name:gsub("ItemSlot_", "")
    if favoredItems[itemID] then return end

    local itemLabel = itemSlot:FindFirstChild("ItemNameTextLabel")
    if itemLabel and itemLabel:IsA("TextLabel") then
        local color = itemLabel.TextColor3
        if isTargetColor(color) then
            local args = { itemID }
            local remote = Network:WaitForChild("RemoteEvents"):WaitForChild("e68ee7f4bd6eb945")
            remote:FireServer(unpack(args))
            favoredItems[itemID] = true
        end
    end
end

DumpTab:CreateToggle({
    Name = "Auto Favor Yellow & Red Items",
    CurrentValue = false,
    Flag = "AutoFavorColors",
    Callback = function(Value)
        isAutoFavoring = Value
        local InventoryFrame = LocalPlayer.PlayerGui:WaitForChild("Backpack")
            :WaitForChild("Inventory")
            :WaitForChild("Container")
            :WaitForChild("ScrollingFrame")

        if Value then
            -- Favor existing items once
            for _, itemSlot in ipairs(InventoryFrame:GetChildren()) do
                if itemSlot:IsA("Frame") and itemSlot.Name:match("ItemSlot_") then
                    favorItem(itemSlot)
                end
            end

            -- Listen for new items added to inventory
            InventoryFrame.ChildAdded:Connect(function(itemSlot)
                if not isAutoFavoring then return end
                if itemSlot:IsA("Frame") and itemSlot.Name:match("ItemSlot_") then
                    -- Wait for the label to exist and fire only once
                    itemSlot:WaitForChild("ItemNameTextLabel")
                    favorItem(itemSlot)
                end
            end)
        else
            -- Reset tracking when toggle is turned off
            favoredItems = {}
        end
    end
})

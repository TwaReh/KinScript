local Rayfield = loadstring(game:HttpGet('https://sirius.menu/rayfield'))()

local Players = game:GetService("Players")
local ReplicatedStorage = game:GetService("ReplicatedStorage")

local player = Players.LocalPlayer

-- STATES
local state = {
    magic = false,
    tp = false,
    rebirth = false,
    egg = false
}

-- REMOTES
local Remotes = ReplicatedStorage:WaitForChild("Remotes")
local magicRemote = Remotes:WaitForChild("GainMagicPower")
local rebirthRemote = Remotes:WaitForChild("Rebirth")
local openEggRemote = Remotes:WaitForChild("OpenEgg")

-- GUI
local Window = Rayfield:CreateWindow({
   Name = "Kin Ja's Hub",
   LoadingTitle = "Kin Ja's Hub",
   LoadingSubtitle = "few games",
   ConfigurationSaving = { Enabled = false }
})

local Tab = Window:CreateTab("+1 Magic Evolution", 4483362458)

-- HRP SAFE
local function getHRP()
    local char = player.Character
    if not char then return nil end
    return char:FindFirstChild("HumanoidRootPart")
end

-- TARGET SAFE
local function getTarget()
    local stages = workspace:FindFirstChild("Stages")
    if not stages then return nil end

    local stage25 = stages:FindFirstChild("Stage25")
    if not stage25 then return nil end

    return stage25:FindFirstChild("WinButton")
end

-- SAFE LOOP WRAPPER (IMPORTANT)
local function runLoop(flagName, func, delay)
    task.spawn(function()
        while state[flagName] do
            local ok, err = pcall(func)
            if not ok then warn(flagName .. " error:", err) end
            task.wait(delay)
        end
    end)
end

-- ======================
-- MAGIC POWER
-- ======================
Tab:CreateToggle({
   Name = "Auto Power",
   CurrentValue = false,

   Callback = function(v)
      state.magic = v

      if v then
         runLoop("magic", function()
            magicRemote:FireServer()
         end, 0.01)
      end
   end
})

-- ======================
-- TELEPORT
-- ======================
Tab:CreateToggle({
   Name = "Auto Win (800B+)",
   CurrentValue = false,

   Callback = function(v)
      state.tp = v

      if v then
         runLoop("tp", function()
            local hrp = getHRP()
            local target = getTarget()

            if hrp and target then
                if target:IsA("BasePart") then
                    hrp.CFrame = target.CFrame + Vector3.new(0, 3, 0)
                elseif target:IsA("Model") then
                    hrp.CFrame = target:GetPivot() + Vector3.new(0, 3, 0)
                end
            end
         end, 0.2)
      end
   end
})

-- ======================
-- REBIRTH
-- ======================
Tab:CreateToggle({
   Name = "Auto Rebirth",
   CurrentValue = false,

   Callback = function(v)
      state.rebirth = v

      if v then
         runLoop("rebirth", function()
            rebirthRemote:FireServer()
         end, 0.5)
      end
   end
})

-- ======================
-- EGG
-- ======================
Tab:CreateToggle({
   Name = "Auto Open Demonic Egg",
   CurrentValue = false,

   Callback = function(v)
      state.egg = v

      if v then
         runLoop("egg", function()
            openEggRemote:InvokeServer("Demonic Egg")
         end, 0.7)
      end
   end
})

local teleportEnabled = false

local function getHRP()
    local char = game.Players.LocalPlayer.Character or game.Players.LocalPlayer.CharacterAdded:Wait()
    return char:WaitForChild("HumanoidRootPart")
end

local function getTarget()
    return workspace:WaitForChild("StaffButtons"):WaitForChild("Staff Button19")
end

Tab:CreateToggle({
   Name = "Auto Best Staff",
   CurrentValue = false,

   Callback = function(Value)
      teleportEnabled = Value

      if teleportEnabled then
         task.spawn(function()
            while teleportEnabled do
                local hrp = getHRP()
                local target = getTarget()

                if hrp and target then
                    if target:IsA("BasePart") then
                        hrp.CFrame = target.CFrame + Vector3.new(0, -3, 0)
                    else
                        hrp.CFrame = target:GetPivot() + Vector3.new(0, -3, 0)
                    end
                end

                task.wait(3) -- 👈 every 5 seconds
            end
         end)
      end
   end
})

local ReplicatedStorage = game:GetService("ReplicatedStorage")

local RuneRemote = ReplicatedStorage:WaitForChild("Remotes"):WaitForChild("RunePickedUp")
local ArmorRemote = ReplicatedStorage:WaitForChild("Remotes"):WaitForChild("ArmorPickedUp")

local selectedItem = "Acid Rune"

Tab:CreateDropdown({
   Name = "Item Giver",
   Options = {
        "Acid Rune",
        "Acid Boots",
        "Acid Plate",
        "Stone Slabs",
        "Molten Rune",
        "Rock Warplate",
        "Vine Rune",
        "Thorned Boots",
        "Beast Hide Armor",
        "Desert Rune",

        "Golden Tiger Guards",
        "Dune Runners",

        "Demonic Rune",
        "Demonic Boots",
        "Demonic Chest",

        "Lunar Rune",
        "Lunar Boots",
        "Lunar Chestplate"
   },

   CurrentOption = "Acid Rune",

   Callback = function(option)
        selectedItem = option[1]
        print("Selected:", selectedItem)
   end
})

Tab:CreateButton({
   Name = "Get Item",
   Callback = function()
        if not selectedItem then return end

        pcall(function()
            if string.find(selectedItem, "Rune") then
                RuneRemote:FireServer(selectedItem)
            else
                ArmorRemote:FireServer(selectedItem)
            end
        end)
   end
})
